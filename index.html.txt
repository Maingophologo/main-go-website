<script>
// --- Storage ---
let parcels = {};
let users = {};
let currentUser = null;
let tempUser = null;
let emailVerificationCode = null;

// --- Show Page ---
function showPage(page){
  document.querySelectorAll("section").forEach(sec=>sec.classList.remove("active"));
  document.getElementById(page).classList.add("active");
}

// --- Simple Password Hash ---
function hashPassword(p){let hash=0;for(let i=0;i<p.length;i++){hash=((hash<<5)-hash)+p.charCodeAt(i);hash=hash & hash;} return hash;}

// --- Registration Verification ---
function sendVerificationCode(){
  let name=document.getElementById("regName").value;
  let email=document.getElementById("regEmail").value;
  let phone=document.getElementById("regPhone").value;
  let pass=document.getElementById("regPass").value;
  if(!name||!email||!phone||!pass){alert("Fill all fields");return;}
  if(users[email]){alert("Email already registered");return;}
  tempUser={name,email,phone,passwordHash:hashPassword(pass)};
  emailVerificationCode=Math.floor(100000+Math.random()*900000);
  alert("For demo purposes, your verification code is: "+emailVerificationCode);
  document.getElementById("verifySection").style.display="block";
}

function verifyEmailCode(){
  let code=document.getElementById("verifyCodeInput").value;
  if(code==emailVerificationCode){
    users[tempUser.email]=tempUser;
    tempUser=null; emailVerificationCode=null;
    document.getElementById("regResult").innerHTML="Registration successful! Please login.";
    document.getElementById("verifySection").style.display="none";
    document.getElementById("regName").value=""; document.getElementById("regEmail").value=""; document.getElementById("regPhone").value=""; document.getElementById("regPass").value=""; document.getElementById("verifyCodeInput").value="";
    showPage('login'); // Phase 2
  } else { alert("Invalid verification code"); }
}

// --- Login ---
function loginUser(){
  let email=document.getElementById("loginEmail").value;
  let pass=document.getElementById("loginPass").value;
  if(users[email] && users[email].passwordHash===hashPassword(pass)){
    currentUser=email;
    document.getElementById("loginResult").innerHTML="Welcome, "+users[email].name+"!";
    showPage('home'); // Phase 3
    document.getElementById("mainNav").style.display="block";
  } else { document.getElementById("loginResult").innerHTML="Invalid email or password"; }
}

// --- Send Parcel ---
function generateTracking(){return "MG"+Math.floor(Math.random()*100000);}
function sendParcel(){
  if(!currentUser){alert("Login first");return;}
  let sender=document.getElementById("sender").value;
  let receiver=document.getElementById("receiver").value;
  let address=document.getElementById("address").value;
  if(!sender||!receiver||!address){alert("Fill all fields");return;}
  let tracking=generateTracking();
  parcels[tracking]={sender,receiver,address,status:"In Transit"};
  document.getElementById("trackingResult").innerHTML="Tracking Number: <b>"+tracking+"</b>";
  document.getElementById("sender").value=""; document.getElementById("receiver").value=""; document.getElementById("address").value="";
}

// --- Track Parcel ---
function trackParcel(){
  let id=document.getElementById("trackID").value;
  if(parcels[id]){document.getElementById("statusResult").innerHTML="Status: <b>"+parcels[id].status+"</b>";}
  else{document.getElementById("statusResult").innerHTML="Parcel not found";}
}
